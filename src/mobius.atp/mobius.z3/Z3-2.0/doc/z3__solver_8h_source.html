00798     template<typename T>
00799     struct coerce {
00800         virtual ~coerce() {}
00801         virtual T* operator()(ast* a) const = 0; 
00802     };
00803 
00804     struct sort_coerce : public coerce<sort> {
00805         virtual sort* operator()(ast* a) const { 
00806             if (!a || a->get_kind() != AST_SORT) {
00807                 return 0;
00808             }
00809             return to_sort(a); 
00810         }
00811     };
00812 
00813     struct func_decl_coerce : public coerce<func_decl> {
00814         virtual func_decl* operator()(ast* a) const { 
00815             if (!a || a->get_kind() != AST_FUNC_DECL) {
00816                 return 0;
00817             }
00818             return to_func_decl(a); 
00819         }
00820     };
00821 
00822     struct pattern_coerce : public coerce<app> {
00823         ast_manager& m_manager;
00824         pattern_coerce(ast_manager& m): m_manager(m) {}
00825         virtual app* operator()(ast* a) const { 
00826             if (!a || a->get_kind() != AST_APP) {
00827                 return 0;
00828             }
00829             if (!m_manager.is_pattern(to_app(a))) {
00830                 return 0;
00831             }
00832             return to_app(a);
00833         }
00834     };
00835 
00836     struct expr_coerce : public coerce<expr> {
00837         virtual expr* operator()(ast* a) const { 
00838             if (!a) {
00839                 return 0;
00840             }
00841             ast_kind k = a->get_kind();
00842             switch(k) {
00843             case AST_APP:
00844             case AST_QUANTIFIER:
00845             case AST_VAR:
00846                 return reinterpret_cast<expr*>(a);
00847             default:
00848                 return 0;
00849             }
00850         }
00851     };

00854     template<typename T>
00855     bool parse_ast(symbol id, T*& n, const coerce<T>& coerce)
00856     {
00857         ast* a;                                                         
00858         if (!m_table.find(id, a)) {
00859             warning_msg("line %d. Could not find id '%s'\n", 
00860                         m_line, id.str().c_str());    
00861             return false;                                               
00862         }                              
00863         n = coerce(a);
00864         if (n == 0) {
00865             warning_msg("line %d. Wrong kind %d for %s\n", 
00866                         m_line, a->get_kind(), id.str().c_str());             
00867             return false;                                               
00868         }                                                               
00869         return true;                                                       
00870     }

00873     template<typename T>
00874     bool parse_ast(T*& n, const coerce<T>& coerce)
00875     {
00876         symbol id;                                                      
00877         if (parse_id(id)) { 
00878             return parse_ast(id, n, coerce);
00879         }                                                                   
00880         return false;                                                       
00881     }

00884     template<typename T>
00885     bool parse_asts(ref_vector<T, ast_manager>& asts, const coerce<T>& c)
00886     {
00887         symbol id;
00888         T*   n;
00889         while (m_string[0] != '\n') {
00890             if (strcmp(m_string.c_str(), "BUILTIN") == 0) {
00891                 return true;
00892             }
00893             if (!parse_ast(n,c)) {
00894                 return false;
00895             }
00896             asts.push_back(n);
00897         }
00898         return true;
00899     }
