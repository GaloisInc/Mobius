let ite_example() =
  begin
    printf "\nite_example\n";
    let ctx         = mk_context [||] in
    let f           = Z3.mk_false ctx in
    let one         = mk_int ctx 1 in
    let zero        = mk_int ctx 0 in
    let ite         = Z3.mk_ite ctx f one zero in
    printf "term: %s\n" (Z3.ast_to_string ctx ite);
    Z3.del_context ctx;
  end
