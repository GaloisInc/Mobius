00211 // Pseudo-predicates are "predicates" defined using the DEFPRED command.
00212 // They are not really predicates, but functions. So, the following commands
00213 // are valid in Simplify:
00214 //
00215 // (DEFPRED (P x))
00216 // (BG_PUSH (EQ (f (P x)) x))
00217 // (BG_PUSH (EQ (P x) (P y)))
00218 // (BG_PUSH (AND (P 0) (P 1)))
00219 //
00220 // In other words, a pseudo-predicate can be used in contexts where a term or fomula
00221 // is expected. When a formula is expected, Simplify adds an equality to |@true|.
00222 // For example, the last command above is mapped to:
00223 //
00224 // (BG_PUSH (AND (EQ (P 0) |@true|) (EQ (P 1) |@true|)))
00225 //
00226 
00227 enum simplify_decl_kind {
00228     SD_FUNCTION,
00229     SD_PREDICATE,
00230     SD_PSEUDO_PREDICATE
00231 };
