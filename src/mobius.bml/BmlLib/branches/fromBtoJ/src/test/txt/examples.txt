assert (forall a b; lo <= a && a < left && lo <= b && b < left && a < b ==> 
          tab[a] <= tab[b]) && 
       (forall a; lo <= a && a < left ==> tab[a] <= tab[left]) && 
       (forall a; left < a && a <= hi ==> tab[left] <= tab[a]) &&
       (forall a; lo <= a && a <= hi ==> 
          (exists a; lo <= a && a <= hi && old(tab[a]) = tab[b]))
